local function AntiGround(Velocity)
    return math.clamp(Velocity.Y * getgenv().Rake.Settings.Prediction, 0.15, math.huge)
end

local function findNearestEnemy()
    local camera = workspace.CurrentCamera
    local localPlayer = Players.LocalPlayer
    local mouseLocation = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y / 2)

    local closestPlayer, closestPart
    local smallestDistance = math.huge

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= localPlayer and player.Character and player.Character:FindFirstChild("Humanoid") and player.Character.Humanoid.Health > 0 then
            local targetPart = player.Character:FindFirstChild(getgenv().Rake.Settings.AimPart)
            if targetPart then
                local screenPos, onScreen = camera:WorldToViewportPoint(targetPart.Position)
                if onScreen then
                    local distFromCrosshair = (Vector2.new(screenPos.X, screenPos.Y) - mouseLocation).Magnitude
                    if distFromCrosshair < smallestDistance then
                        smallestDistance = distFromCrosshair
                        closestPlayer = player
                        closestPart = targetPart
                    end
                end
            end
        end
    end

    return closestPlayer, closestPart
end



local function highlight(plr)
    if plr and plr.Character then
        for _, player in pairs(game.Players:GetPlayers()) do
            if player.Character then
                for _, obj in pairs(player.Character:GetChildren()) do
                    if obj:IsA("Highlight") then
                        obj:Destroy()
                    end
                end
            end
        end
        for _, obj in pairs(plr.Character:GetChildren()) do
            if obj:IsA("Highlight") then
                obj:Destroy()
            end
        end
        local highlight = Instance.new("Highlight")
        highlight.Parent = plr.Character
        highlight.FillColor = Color3.new(0.411765, 0.501961, 1.000000)
        highlight.OutlineColor = Color3.new(0.031373, 0.031373, 0.031373)
        highlight.FillTransparency = 0.6
        highlight.OutlineTransparency = 0
    end
end

 RunService.Heartbeat:Connect(function()
    if getgenv().Rake.Settings.Misc.ForceHit and enabled then
        Plr, ClosestPart = findNearestEnemy() -- constantly update target

        if ClosestPart and Plr and Plr.Character and not Plr.Character:FindFirstChildOfClass("ForceField") then
            local LocalPlayer = Players.LocalPlayer
            local CurrentPosition = LocalPlayer.Character.HumanoidRootPart.Position
            local ShootDirection = LocalPlayer.Character.HumanoidRootPart.CFrame.LookVector
            local ShootPosition = CurrentPosition + ShootDirection * 10
            local Normal = ShootDirection.Unit
            local Offset = Normal * 0.5

            local Args = {
                [1] = "Shoot",
                [2] = {
                    [1] = {
                        [1] = {
                            ["Instance"] = ClosestPart,
                            ["Normal"] = Normal,
                            ["Position"] = CurrentPosition
                        }
                    },
                    [2] = {
                        [1] = {
                            ["thePart"] = ClosestPart,
                            ["theOffset"] = CFrame.new(Offset)
                        }
                    },
                    [3] = ShootPosition,
                    [4] = CurrentPosition,
                    [5] = tick()
                }
            }

            ReplicatedStorage.MainEvent:FireServer(unpack(Args))
        end
    end
end)
